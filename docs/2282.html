左外部连接

SELECT *
FROM Table1 t1
LEFT OUTER JOIN Table2 t2
ON t1.fk = t2.id;

半连接

SELECT *
FROM Table1 t1
WHERE EXISTS (SELECT 1FROM Table2 t2
WHERE t1.fk = t2.id

FROM Table1 t1
LEFT OUTER JOIN Table2 t2
ON t1.fk = t2.id

内部连接

右外部联接

反半连接

SELECT *
FROM Table1 t1
WHERE NOT EXISTS (SELECT 1FROM Table2 t2
WHERE t1.fk = t2.id

排除的右外连接

SELECT *
FROM Table1 t1
RIGHT OUTER JOIN Table2 t2
ON t1.fk = t2.id
WHERE t1.fk IS NULL;

交叉连接-笛卡尔乘积

SELECT *
FROM Table1 t1
CROSS JOIN Table2 t2;

非对等内部连接

SELECT *
FROM Table1 t1
INNER JOIN Table2 t2
ON t1.fk >= t2.id;

交叉应用

SELECT *
FROM Table1 t1
CROSS APPLY[dbo].[someTVF](t1.fk)
AS t;

外部应用

SELECT *
FROM Table1 t1
OUTER APPLY[dbo].[someTVF](t1.fk)
AS t;
